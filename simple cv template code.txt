    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // Form elements
            const form = document.getElementById('cv-form');
            const formContainer = document.getElementById('form-container');
            const cvDisplay = document.getElementById('cv-display');
            const toggleViewBtn = document.getElementById('toggle-view-btn');
            let isViewingCV = false;

            // Toggle between form and CV view
            toggleViewBtn.addEventListener('click', function() {
                isViewingCV = !isViewingCV;
                
                if (isViewingCV) {
                    formContainer.style.display = 'none';
                    cvDisplay.style.display = 'block';
                    toggleViewBtn.innerHTML = '<i class="fas fa-edit"></i>';
                } else {
                    formContainer.style.display = 'block';
                    cvDisplay.style.display = 'none';
                    toggleViewBtn.innerHTML = '<i class="fas fa-eye"></i>';
                }
            });

            // Initially show the form
            formContainer.style.display = 'block';
            cvDisplay.style.display = 'none';

            // Add skill field
            document.getElementById('add-skill').addEventListener('click', function() {
                const container = document.getElementById('skills-container');
                const div = document.createElement('div');
                div.className = 'skill-input';
                div.style.marginBottom = '0.75rem';
                div.innerHTML = `
                    <input type="text" class="skill" placeholder="Skill">
                    <button type="button" class="btn btn-remove remove-skill"><i class="fas fa-minus"></i></button>
                `;
                container.appendChild(div);
            });

            // Remove skill field
            document.addEventListener('click', function(e) {
                if (e.target.classList.contains('remove-skill') || 
                    (e.target.parentElement && e.target.parentElement.classList.contains('remove-skill'))) {

            // Form submission
            form.addEventListener('submit', function(e) {
                e.preventDefault();
                
                // Basic info
                document.getElementById('cv-name').textContent = document.getElementById('full-name').value;
                document.getElementById('cv-title').textContent = document.getElementById('job-title').value;
                document.getElementById('cv-about').textContent = document.getElementById('about-me').value;
                
                // Photo
                const photoUrl = document.getElementById('profile-photo').value;
                if (photoUrl) {
                    document.getElementById('cv-photo').src = photoUrl;
                } else {
                    document.getElementById('cv-photo').src = 'https://via.placeholder.com/150';
                }
                
                // Contact info
                document.getElementById('cv-email').textContent = document.getElementById('email').value;
                document.getElementById('cv-phone').textContent = document.getElementById('phone').value;
                document.getElementById('cv-address').textContent = document.getElementById('address').value;
                
                // Skills
                const skillsContainer = document.getElementById('cv-skills');
                skillsContainer.innerHTML = '';
                document.querySelectorAll('.skill').forEach(skillInput => {
                    if (skillInput.value) {
                        const skill = document.createElement('span');
                        skill.className = 'skill';
                        skill.textContent = skillInput.value;
                        skillsContainer.appendChild(skill);
                    }
                });
                
                // Education
                const educationContainer = document.getElementById('cv-education');
                educationContainer.innerHTML = '';
                document.querySelectorAll('.education-item').forEach(eduItem => {
                    const institution = eduItem.querySelector('.institution').value;
                    const degree = eduItem.querySelector('.degree').value;
                    const duration = eduItem.querySelector('.duration').value;
                    
                    if (institution && degree && duration) {
                        const div = document.createElement('div');
                        div.className = 'timeline-item';
                        div.innerHTML = `
                            <div class="timeline-date"><i class="far fa-calendar-alt"></i> ${duration}</div>
                            <h3>${institution}</h3>
                            <p>${degree}</p>
                        `;
                        educationContainer.appendChild(div);
                    }
                });
                
                // Experience
                const experienceContainer = document.getElementById('cv-experience');
                experienceContainer.innerHTML = '';
                document.querySelectorAll('.experience-item').forEach(expItem => {
                    const jobTitle = expItem.querySelector('.job-title').value;
                    const company = expItem.querySelector('.company').value;
                    const duration = expItem.querySelector('.duration').value;
                    const responsibilities = expItem.querySelector('.responsibilities').value;
                    
                    if (jobTitle && company && duration) {
                        const div = document.createElement('div');
                        div.className = 'timeline-item';
                        
                        let responsibilitiesHtml = '';
                        if (responsibilities) {
                            responsibilitiesHtml = `<ul style="margin-left: 1.5rem; margin-top: 0.75rem; line-height: 1.8;">`;
                            responsibilities.split('\n').forEach(resp => {
                                if (resp.trim()) {
                                    responsibilitiesHtml += `<li>${resp}</li>`;
                                }
                            });
                            responsibilitiesHtml += `</ul>`;
                        }
                        
                        div.innerHTML = `
                            <div class="timeline-date"><i class="far fa-calendar-alt"></i> ${duration}</div>
                            <h3>${jobTitle}</h3>
                            <p>${company}</p>
                            ${responsibilitiesHtml}
                        `;
                        experienceContainer.appendChild(div);
                    }
                });
                
                // Projects
                const projectsContainer = document.getElementById('cv-projects');
                projectsContainer.innerHTML = '';
                document.querySelectorAll('.project-item').forEach(projItem => {
                    const name = projItem.querySelector('.project-name').value;
                    const description = projItem.querySelector('.project-description').value;
                    
                    if (name && description) {
                        const div = document.createElement('div');
                        div.className = 'timeline-item';
                        div.innerHTML = `
                            <h3>${name}</h3>
                            <p>${description}</p>
                        `;
                        projectsContainer.appendChild(div);
                    }
                });
                
                // Social links
                const socialLinksContainer = document.getElementById('cv-social-links');
                socialLinksContainer.innerHTML = '';
                document.querySelectorAll('.social-link-input').forEach(socialItem => {
                    const type = socialItem.querySelector('.social-type').value;
                    const url = socialItem.querySelector('.social-url').value;
                    
                    if (url) {
                        let iconClass = '';
                        switch(type) {
                            case 'linkedin': iconClass = 'fab fa-linkedin-in'; break;
                            case 'github': iconClass = 'fab fa-github'; break;
                            case 'twitter': iconClass = 'fab fa-twitter'; break;
                            case 'instagram': iconClass = 'fab fa-instagram'; break;
                            default: iconClass = 'fas fa-link';
                        }
                        
                        const link = document.createElement('a');
                        link.className = 'social-link';
                        link.href = url;
                        link.target = '_blank';
                        link.innerHTML = `<i class="${iconClass}"></i>`;
                        socialLinksContainer.appendChild(link);
                    }
                });
                
                // Show the CV with animation
                formContainer.style.display = 'none';
                cvDisplay.style.display = 'block';
                isViewingCV = true;
                toggleViewBtn.innerHTML = '<i class="fas fa-edit"></i>';
                
                // Add animation classes to CV elements
                const cvElements = cvDisplay.querySelectorAll('section, .timeline-item, .skill');
                cvElements.forEach((el, index) => {
                    el.classList.add('animate');
                    el.style.animationDelay = `${index * 0.1}s`;
                });
            });
        });
